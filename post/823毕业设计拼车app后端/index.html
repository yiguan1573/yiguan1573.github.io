数据库保存emoji表情

  • 问题描述：java.sql.SQLException: Incorrect string value: ‘\xF0\x9F\x92\xAA”,……’ for column ‘raw_json’ at row 1, 异常：org.springframework.jdbc.UncategorizedSQLException:
  • 解决：普通的字符串或者表情都是占位3个字节，移动端的表情符号占位是4个字节，普通的utf8就不够用了，MySQL 数据库建议都提前采用utf8mb4字符集。将数据库，表，字段都改为utf8mb4编码，然后重启数据库。如果表结构复杂，就只能前端过滤emoji表情，代码如下
sendMessage(n){
	let l = n.replace(/\s*/g,"")
	//过滤emoji表情
	let regStr = /[\uD83C|\uD83D|\uD83E][\uDC00-\uDFFF][\u200D|\uFE0F]|[\uD83C|\uD83D|\uD83E][\uDC00-\uDFFF]|[0-9|*|#]\uFE0F\u20E3|[0-9|#]\u20E3|[\u203C-\u3299]\uFE0F\u200D|[\u203C-\u3299]\uFE0F|[\u2122-\u2B55]|\u303D|[\A9|\AE]\u3030|\uA9|\uAE|\u3030/ig;
	
	if(l == ""){
		this.$refs.uToast.show({
							title: '输入不能为空',
							type: 'error',
							duration: 1000
						})
	}else if(uni.getStorageSync('token') == ""){
		this.$refs.uToast.show({
							title: '请先登录',
							type: 'error',
							url: '/pages/login/login',
							duration: 1000
						})
	}else if(this.sendMark == 1){
		let item = this.value
		if(regStr.test(item)){
			item = item.replace(regStr,'')//过滤emoji表情
		}
		this.$emit('send1',item)
		this.value = ""
	}else if(this.sendMark == 2){
		let item = this.value
		if(regStr.test(item)){
			item = item.replace(regStr,'')//过滤emoji表情
		}
		this.$emit('send2',item)
		this.value = ""
	}
}
