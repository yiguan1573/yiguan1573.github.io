public class ZKTest {
    public static void main(String[] args) {
        ZooKeeper zooKeeper  = null;
        //zookeeper获取连接是一个慢启动的过程
        CountDownLatch countDownLatch = new CountDownLatch(1);
        try {
            zooKeeper = new ZooKeeper("192.168.10.101:2181", 3000, new Watcher() {
                @Override
                public void process(WatchedEvent watchedEvent) {
                    Event.KeeperState state = watchedEvent.getState();
                    if(Event.KeeperState.SyncConnected.equals(state)){
                        System.out.println("连接成功");
                        countDownLatch.countDown();
                    }else if(Event.KeeperState.Closed.equals(state)){
                        System.out.println("关闭连接");
                    }
                }
            });
            countDownLatch.await();
            //操作zookeeper
             zooKeeper.create("/yiguan/test1","hello".getBytes(), ZooDefs.Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
        } catch (IOException e) {
            e.printStackTrace();
        } catch (InterruptedException e) {
            e.printStackTrace();
        } finally {
            if(zooKeeper!=null){
                try {
                    zooKeeper.close();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}

@Component
public class ZkClient {

    private ZooKeeper zooKeeper;

    @PostConstruct
    public void init(){
        CountDownLatch countDownLatch = new CountDownLatch(1);
        //项目启动时，获取连接
        try {
            zooKeeper = new ZooKeeper("192.168.10.101:2181", 3000, new Watcher() {
                @Override
                public void process(WatchedEvent watchedEvent) {
                    Event.KeeperState state = watchedEvent.getState();
                    if(Event.KeeperState.SyncConnected.equals(state)&&Event.EventType.None.equals(watchedEvent.getType())){
                        System.out.println("连接成功:"+watchedEvent);
                        countDownLatch.countDown();
                    }else if(Event.KeeperState.Closed.equals(state)){
                        System.out.println("关闭连接");
                    }else{
                        System.out.println("节点事件");
                    }
                }
            });
            countDownLatch.await();
        } catch (IOException e) {
            e.printStackTrace();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    @PreDestroy
    public void destroy(){
        //释放zk连接
        if(zooKeeper!=null){
            try {
                zooKeeper.close();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }

    public DistributedZkLock getLock(String lock){
        return new DistributedZkLock(lock,zooKeeper);
    }
}

public class DistributedZkLock implements Lock {
    private String lockName;
    private ZooKeeper zooKeeper;

    private static final  String ROOT_PATH = "/locks";

    public DistributedZkLock(String lockName, ZooKeeper zooKeeper) {
        this.lockName = ROOT_PATH+"/"+lockName;
        this.zooKeeper = zooKeeper;
        try {
            if(zooKeeper.exists(ROOT_PATH,false) == null){
                zooKeeper.create(ROOT_PATH,null, ZooDefs.Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
            }
        } catch (KeeperException e) {
            e.printStackTrace();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    @Override
    public void lock() {
        tryLock();
    }

    @Override
    public void lockInterruptibly() throws InterruptedException {

    }

    @Override
    public boolean tryLock() {
        //创建znode节点过程:为了防止zk客户端程序获取到锁之后，服务器宕机带来的死锁问题，这里创建的是临时节点
        try {
            zooKeeper.create(lockName,null,ZooDefs.Ids.OPEN_ACL_UNSAFE,CreateMode.EPHEMERAL);
            return true;
        } catch (Exception e) {
            e.printStackTrace();
            try {
                Thread.sleep(40);
                tryLock();
            } catch (InterruptedException e1) {
                e1.printStackTrace();

            }
        }
        return false;
    }

    @Override
    public boolean tryLock(long time, TimeUnit unit) throws InterruptedException {
        return false;
    }

    @Override
    public void unlock() {
        //删除znode
        try {
            zooKeeper.delete(lockName,-1);
        } catch (InterruptedException e) {
            e.printStackTrace();
        } catch (KeeperException e) {
            e.printStackTrace();
        }

    }

    @Override
    public Condition newCondition() {
        return null;
    }
}

锁优化之阻塞锁

  • 临时序列化节点实现阻塞锁,创建节点后判断自己是不是最小的节点，如果是就获取到锁，如果不是就监听比自己小的哪一个序列化节点是否被删除，被删除了，自己就是最小的了，就获取到锁
  • 该方法同时也实现了公平锁 
public class DistributedZkLock implements Lock {
    private String lockName;
    private ZooKeeper zooKeeper;
    private String currentNodePath;
    private String fullPath;

    private static final  String ROOT_PATH = "/locks";

    public DistributedZkLock(String lockName, ZooKeeper zooKeeper) {
        this.fullPath = ROOT_PATH+"/"+lockName+"-";
        this.lockName = lockName;
        this.zooKeeper = zooKeeper;
        try {
            if(zooKeeper.exists(ROOT_PATH,false) == null){
                zooKeeper.create(ROOT_PATH,null, ZooDefs.Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
            }
        } catch (KeeperException e) {
            e.printStackTrace();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    @Override
    public void lock() {
        tryLock();
    }

    @Override
    public void lockInterruptibly() throws InterruptedException {

    }

    @Override
    public boolean tryLock() {
        //创建znode节点过程:为了防止zk客户端程序获取到锁之后，服务器宕机带来的死锁问题，这里创建的是临时节点
        try {
            currentNodePath = zooKeeper.create(fullPath, null, ZooDefs.Ids.OPEN_ACL_UNSAFE, CreateMode.EPHEMERAL_SEQUENTIAL);
            //获取前置节点，如果前置节点为空，则获取锁成功，否则监听前置节点
            String preNode = getPreNode();
            if(preNode != null){
                //利用闭锁思想，实现阻塞功能
                CountDownLatch countDownLatch = new CountDownLatch(1);
                //因为获取前置节点这个操作，不具备原子性。再次判断zk中的前置节点是否存在
                if(zooKeeper.exists(ROOT_PATH + "/" + preNode, new Watcher() {
                    @Override
                    public void process(WatchedEvent watchedEvent) {
                        countDownLatch.countDown();
                    }
                }) == null){
                    return true;
                }
                countDownLatch.await();
            }
            return true;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return false;
    }

    private String getPreNode() {
        try {
            //获取根节点下的所有节点
            List<String> children = zooKeeper.getChildren(ROOT_PATH, false);
            if(CollectionUtils.isEmpty(children)){
                throw new IllegalMonitorStateException("非法操作");
            }
            //获取当前节点同一资源的锁
            List<String> nodes = children.stream().filter(e -> e.startsWith(lockName+"-")).collect(Collectors.toList());
            if(CollectionUtils.isEmpty(nodes)){
                throw new IllegalMonitorStateException("非法操作");
            }
            //排队
            Collections.sort(nodes);
            //获取当前节点的下标
            String currentNode = StringUtils.substringAfterLast(currentNodePath, "/");//当前节点名
            int index = Collections.binarySearch(nodes, currentNode);
            if(index < 0){
                throw new IllegalMonitorStateException("非法操作");
            }else if(index > 0){
                return nodes.get(index - 1);//返回前置节点
            }else {
                //如果当前节点就是第一个节点，则返回null
                return null;
            }
        } catch (Exception e) {
            e.printStackTrace();
            throw new IllegalMonitorStateException("非法操作");
        }
    }

    @Override
    public boolean tryLock(long time, TimeUnit unit) throws InterruptedException {
        return false;
    }

    @Override
    public void unlock() {
        //删除znode
        try {
            zooKeeper.delete(currentNodePath,-1);
        } catch (InterruptedException e) {
            e.printStackTrace();
        } catch (KeeperException e) {
            e.printStackTrace();
        }

    }

    @Override
    public Condition newCondition() {
        return null;
    }
}

  • 代码
public class DistributedZkLock implements Lock {
    private String lockName;
    private ZooKeeper zooKeeper;
    private String currentNodePath;
    private String fullPath;
    private static final ThreadLocal<Integer> THREAD_LOCAL = new ThreadLocal<>();

    private static final  String ROOT_PATH = "/locks";

    public DistributedZkLock(String lockName, ZooKeeper zooKeeper) {
        this.fullPath = ROOT_PATH+"/"+lockName+"-";
        this.lockName = lockName;
        this.zooKeeper = zooKeeper;
        try {
            if(zooKeeper.exists(ROOT_PATH,false) == null){
                zooKeeper.create(ROOT_PATH,null, ZooDefs.Ids.OPEN_ACL_UNSAFE, CreateMode.PERSISTENT);
            }
        } catch (KeeperException e) {
            e.printStackTrace();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    @Override
    public void lock() {
        tryLock();
    }

    @Override
    public void lockInterruptibly() throws InterruptedException {

    }

    @Override
    public boolean tryLock() {
        try {
            //判断threadLocal中是否已经有锁，有锁直接重入(+1)
            Integer flag = THREAD_LOCAL.get();
            if(flag!=null&&flag>0){
                THREAD_LOCAL.set(flag+1);
                return true;
            }
            //创建znode节点过程:为了防止zk客户端程序获取到锁之后，服务器宕机带来的死锁问题，这里创建的是临时节点
            currentNodePath = zooKeeper.create(fullPath, null, ZooDefs.Ids.OPEN_ACL_UNSAFE, CreateMode.EPHEMERAL_SEQUENTIAL);
            //获取前置节点，如果前置节点为空，则获取锁成功，否则监听前置节点
            String preNode = getPreNode();
            if(preNode != null){
                //利用闭锁思想，实现阻塞功能
                CountDownLatch countDownLatch = new CountDownLatch(1);
                //因为获取前置节点这个操作，不具备原子性。再次判断zk中的前置节点是否存在
                if(zooKeeper.exists(ROOT_PATH + "/" + preNode, new Watcher() {
                    @Override
                    public void process(WatchedEvent watchedEvent) {
                        countDownLatch.countDown();
                    }
                }) == null){
                    THREAD_LOCAL.set(1);
                    return true;
                }
                countDownLatch.await();
            }
            THREAD_LOCAL.set(1);
            return true;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return false;
    }

    private String getPreNode() {
        try {
            //获取根节点下的所有节点
            List<String> children = zooKeeper.getChildren(ROOT_PATH, false);
            if(CollectionUtils.isEmpty(children)){
                throw new IllegalMonitorStateException("非法操作");
            }
            //获取当前节点同一资源的锁
            List<String> nodes = children.stream().filter(e -> e.startsWith(lockName+"-")).collect(Collectors.toList());
            if(CollectionUtils.isEmpty(nodes)){
                throw new IllegalMonitorStateException("非法操作");
            }
            //排队
            Collections.sort(nodes);
            //获取当前节点的下标
            String currentNode = StringUtils.substringAfterLast(currentNodePath, "/");//当前节点名
            int index = Collections.binarySearch(nodes, currentNode);
            if(index < 0){
                throw new IllegalMonitorStateException("非法操作");
            }else if(index > 0){
                return nodes.get(index - 1);//返回前置节点
            }else {
                //如果当前节点就是第一个节点，则返回null
                return null;
            }
        } catch (Exception e) {
            e.printStackTrace();
            throw new IllegalMonitorStateException("非法操作");
        }
    }

    @Override
    public boolean tryLock(long time, TimeUnit unit) throws InterruptedException {
        return false;
    }

    @Override
    public void unlock() {
        //删除znode
        try {
            THREAD_LOCAL.set(THREAD_LOCAL.get() - 1);
            if(THREAD_LOCAL.get() == 0){
                zooKeeper.delete(currentNodePath,-1);
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        } catch (KeeperException e) {
            e.printStackTrace();
        }

    }

    @Override
    public Condition newCondition() {
        return null;
    }
}

InterProcessMutex可重入锁

  • InterProcessMutex(CuratorFramework client, String paths)
  • 代码
 public void reduce(){
    InterProcessMutex multiLock = new InterProcessMutex(curatorFramework, "/curator/locks");
    try {
        multiLock.acquire();
        //1.查询库存
        Integer stock = (Integer) redisTemplate.opsForValue().get("stock");

        //2.判断库存是否充足
        if(stock != null&& stock> 0){
            //3.扣减库存
            redisTemplate.opsForValue().set("stock",stock-1);
        }
    } catch (Exception e) {
        e.printStackTrace();
    } finally {
        try {
            multiLock.release();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

InterProcessSemaphoreMutex不可重入锁

  • 使用与InterProcessMutex相同

InterProcessReadWriteLock可重入的读写锁

  • 使用
InterProcessReadWriteLock readWriteLock = new InterProcessReadWriteLock(curatorFramework, "/curator/locks");
readWriteLock.readLock().acquire();
readWriteLock.readLock().release();
readWriteLock.writeLock().acquire();
readWriteLock.writeLock().release();
  • 读读可以并发的
  • 读写不可以并发
  • 写写不可以并发
  • 写锁在释放之前会阻塞请求钱程,而读锁是不会的。
