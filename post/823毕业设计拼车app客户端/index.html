正则表达式验证密码问题

  • 问题：js正则表达式test第一次为true，第二次为false
  • 解决办法：重新创建一个正则表达式
  • 原因：为什么使用正则test( )第一次是 true,第二次是false？
sendMessage(n1,n2,n3) {
    let l1 = n1.replace(/\s*/g,"")
    let l2 = n2.replace(/\s*/g,"")
    let l3 = n3.replace(/\s*/g,"")
    let reg=/^[A-Za-z]+[0-9]+[A-Za-z0-9]*|[0-9]+[A-Za-z]+[A-Za-z0-9]*$/g
    let reg1=/^[A-Za-z]+[0-9]+[A-Za-z0-9]*|[0-9]+[A-Za-z]+[A-Za-z0-9]*$/g
    if(l1 == "" || l2 == "" || l3 == ""){
        this.$refs.uTips.show({
                        title: '密码不能为空',
                        type: 'error',
                        duration: '1000'
                    })
    }else if(l2 != l3){
        this.$refs.uTips.show({
                        title: '密码不一致',
                        type: 'error',
                        duration: '1000'
                    })
    }else if(!reg.test(n1)){
        this.$refs.uTips.show({
                        title: '密码必须由6-16个英文字母和数字的字符串组成！',
                        type: 'error',
                        duration: '1000'
                    })
    }else if(!reg1.test(n3)){
        this.$refs.uTips.show({
                        title: '密码必须由6-16个英文字母和数字的字符串组成！',
                        type: 'error',
                        duration: '1000'
                    })
    
    }else {
        //上传服务器
    }

data() {

	}

uniapp下拉刷新无法返回

  • 代码如下
onPullDownRefresh() {
	// that = this
	uni.startPullDownRefresh()
	uni.reLaunch({
			url: '/pages/myCarpool/myCarpool'
	});
	uni.stopPullDownRefresh()
	
}
  • uni.reLaunch会关闭所有页面，打开到应用内的某个页面，就不会有返回
  • uni.redirectTo关闭当前页面，跳转到应用内的某个页面
  • 但是在tabar页面使用uni.redirectTo无效，且tabar页面页面不需要返回，正好可以使用uni.reLaunch
  • 结论：tabar页面uni.reLaunch，其他页面使用uni.redirectTo

uniapp聊天室滚动到底部

  • 问题描述：每次发送或者收到一个消息时，页面需要滚动到最底部查看消息
  • 解决办法：如下
setTimeout(() => {
	uni.pageScrollTo({scrollTop: 99999, duration: 0});
}, 50);
//如果uni.pageScrollTo不生效，在最外层wrapper设置 height:auto!important
  • 完整代码
<template>
	<view>
		<inputBox @send1="comment"></inputBox>
		<view class="messageList">
			<view  v-for="(item,index) in chatData" :key="index">
				<chatMessageLeft :otherChatItemData="item" v-if="item.isLeft"></chatMessageLeft>
				<chatMessageRight :myChatItemData="item" v-if="!item.isLeft"></chatMessageRight>
			</view>	
			<view class="messageBox"></view>
		</view> 
							<!-- uni.pageScrollTo()	将页面滚动到目标位置 -->
		<u-no-network></u-no-network>
	</view>
</template>

<script>
	export default {
		onLoad() {
			uni.connectSocket({
				url: 'http://192.168.43.90:8888/serve/1'
			});
			uni.onSocketOpen( res => {
				this.socketOpen = true;
				 console.log('WebSocket连接已打开！');
			});
			uni.onSocketError(function (res) {
			  console.log('WebSocket连接打开失败，请检查！');
			});
			uni.onSocketMessage( res => {
			  console.log('收到服务器内容：' + res.data);
				this.chatData.push(JSON.parse(res.data))
				setTimeout(() => {
				   uni.pageScrollTo({scrollTop: 99999, duration: 0});
				}, 50);
			})
			uni.onSocketClose(function(res) {
				console.log('WebSocket 已关闭！');
			});
		},
		data() {
			return {
				socketOpen:false,
				chatData: []
			};
		},
		methods:{
			comment(e){
				if (this.socketOpen) {
					let msg = {
						toUserId: 2,
						contentText: e,
						isLeft: false
						// 判断消息是左边还是右边
					}
					this.chatData.push(msg)
					setTimeout(() => {
					   uni.pageScrollTo({scrollTop: 99999, duration: 0});
					}, 50);
					uni.sendSocketMessage({
						data: JSON.stringify(msg)
					});
				}
			}
		},
		onUnload(){
				//关闭socket
				uni.closeSocket()
		}
	}
</script>
